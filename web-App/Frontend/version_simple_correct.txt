  <h2>Uploader une image</h2>
  <input type="file" id="imageInput">
  <button onclick="sendImage()">Prédire</button>

  <p id="result"></p>

  <!-- Lecteur audio masqué au départ -->
  <audio id="player" controls autoplay hidden></audio>

  <script>
    async function sendImage() {
      const input = document.getElementById("imageInput");
      const file = input.files[0];
      if (!file) {
        alert("Choisissez une image !");
        return;
      }

      const formData = new FormData();
      formData.append("file", file);

      try {
        const response = await fetch("http://127.0.0.1:8000/predict", {
          method: "POST",
          body: formData
        });

        const result = await response.json();

        document.getElementById("result").innerText =
          `Classe prédite : ${result.predicted_class} (confiance : ${result.confidence}%)`;

        // Jouer le fichier audio associé
        const audio = document.getElementById("player");
        audio.src = "http://127.0.0.1:8000" + result.audio_url;
        audio.hidden = false;
        audio.play();

      } catch (error) {
        console.error(error);
        document.getElementById("result").innerText =
          "Erreur lors de la prédiction.";
      }
    }
  </script>
</body>
